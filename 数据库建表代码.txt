-- Table: public.adoptions

-- DROP TABLE IF EXISTS public.adoptions;

CREATE TABLE IF NOT EXISTS public.adoptions
(
    id integer NOT NULL DEFAULT nextval('adoptions_id_seq'::regclass),
    user_id integer,
    animal_id integer,
    college character varying COLLATE pg_catalog."default",
    student_class character varying COLLATE pg_catalog."default",
    apply_reason character varying COLLATE pg_catalog."default",
    status character varying COLLATE pg_catalog."default",
    admin_comment character varying COLLATE pg_catalog."default",
    apply_date timestamp without time zone,
    CONSTRAINT adoptions_pkey PRIMARY KEY (id)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.adoptions
    OWNER to postgres;
-- Index: ix_adoptions_animal_id

-- DROP INDEX IF EXISTS public.ix_adoptions_animal_id;

CREATE INDEX IF NOT EXISTS ix_adoptions_animal_id
    ON public.adoptions USING btree
    (animal_id ASC NULLS LAST)
    WITH (fillfactor=100, deduplicate_items=True)
    TABLESPACE pg_default;
-- Index: ix_adoptions_id

-- DROP INDEX IF EXISTS public.ix_adoptions_id;

CREATE INDEX IF NOT EXISTS ix_adoptions_id
    ON public.adoptions USING btree
    (id ASC NULLS LAST)
    WITH (fillfactor=100, deduplicate_items=True)
    TABLESPACE pg_default;
